Secret Message

Think about a thing. You want to sent a message to your friend trying no one will understand it. The theory is your message will be encrypted by the opposite corresponding letters in the English alphabet. Like, if you enter A, your massage will be Z. So let's make it..

#include <iostream>
            using namespace std;

            void lowercase(string& str){
            int len=str.length();
            int i;
            for(i=0;i<=len-1;i++){
            if(str[i]<='Z'&&str[i]>='A'){
                str[i]=str[i]+32;
            }
            }
            }
            void encrypt(string& str){
            int len=str.length();
            int i;
            for(i=0;i<=len-1;i++){
                if(str[i]==' ')continue;
                str[i]=219-str[i];
            }
            }
            int main()
            {
            string msg;
            getline(cin,msg);
            lowercase(msg);
            encrypt(msg);
            cout<<msg<<endl;
            return 0;
            }
